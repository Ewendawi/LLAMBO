# Description
**AdaptiveParetoLCB_EI_TrustRegionBO (APLTEI-TRBO):** This algorithm combines the strengths of AdaptiveParetoLCBTrustRegionBO and AdaptiveParetoTrustRegionBO by utilizing both LCB and EI in the Pareto selection. It adaptively switches between LCB and EI based on the optimization progress, favoring LCB initially for exploration and EI later for exploitation. A dynamic weighting mechanism adjusts the influence of LCB and EI in the Pareto front construction. The algorithm also incorporates a mechanism to dynamically adjust the kappa parameter in the LCB calculation, further enhancing the exploration-exploitation balance.

# Justification
The rationale behind this algorithm is to leverage the complementary strengths of LCB and EI. LCB is known for its exploration capabilities, particularly in the early stages of optimization, while EI excels at exploitation as the search progresses. By adaptively switching between or combining these acquisition functions, the algorithm can more effectively navigate the search space and converge to the global optimum.

The dynamic weighting of LCB and EI allows the algorithm to fine-tune the exploration-exploitation balance based on the optimization progress. Initially, LCB might be given a higher weight to encourage exploration of the search space. As the algorithm identifies promising regions, the weight of EI can be increased to focus on exploiting these regions.

The dynamic adjustment of the kappa parameter in the LCB calculation provides another layer of control over the exploration-exploitation balance. A higher kappa value promotes exploration by increasing the penalty for uncertainty, while a lower kappa value encourages exploitation by focusing on the predicted mean.

# Code
```python
from collections.abc import Callable
from scipy.stats import qmc
from scipy.stats import norm
import numpy as np
from sklearn.gaussian_process import GaussianProcessRegressor
from sklearn.gaussian_process.kernels import RBF, ConstantKernel
from scipy.spatial.distance import cdist

class AdaptiveParetoLCB_EI_TrustRegionBO:
    def __init__(self, budget: int, dim: int):
        self.budget = budget
        self.dim = dim
        self.bounds = np.array([[-5.0] * dim, [5.0] * dim])
        self.X: np.ndarray = None
        self.y: np.ndarray = None
        self.n_evals = 0
        self.n_init = min(5 * dim, self.budget // 10)
        self.gp = None
        self.best_x = None
        self.best_y = float('inf')
        self.trust_region_radius = 2.0
        self.radius_decay = 0.95
        self.radius_increase = 1.1
        self.min_radius = 0.1
        self.success_ratio = 0.0
        self.random_restart_prob = 0.05
        self.kappa = 2.0  # Initial kappa value for LCB
        self.kappa_decay = 0.995
        self.ei_weight = 0.1 # Initial weight for EI
        self.ei_weight_increase = 1.01

    def _sample_points(self, n_points):
        if self.best_x is None or np.random.rand() < self.random_restart_prob:
            sampler = qmc.LatinHypercube(d=self.dim)
            sample = sampler.random(n=n_points)
            return qmc.scale(sample, self.bounds[0], self.bounds[1])
        else:
            points = []
            while len(points) < n_points:
                sample = np.random.uniform(-1, 1, size=(self.dim,))
                sample = self.best_x + self.trust_region_radius * sample
                sample = np.clip(sample, self.bounds[0], self.bounds[1])
                points.append(sample)
            return np.array(points)

    def _fit_model(self, X, y):
        kernel = ConstantKernel(1.0, constant_value_bounds="fixed") * RBF(length_scale=1.0, length_scale_bounds="fixed")
        self.gp = GaussianProcessRegressor(kernel=kernel, n_restarts_optimizer=0, alpha=1e-6)
        self.gp.fit(X, y)
        return self.gp

    def _expected_improvement(self, X):
        if self.gp is None:
            return np.random.normal(size=(len(X), 1))
        else:
            mu, sigma = self.gp.predict(X, return_std=True)
            sigma = np.clip(sigma, 1e-9, np.inf)
            imp = self.best_y - mu
            Z = imp / sigma
            ei = imp * norm.cdf(Z) + sigma * norm.pdf(Z)
            return ei.reshape(-1, 1)

    def _lower_confidence_bound(self, X):
        if self.gp is None:
            return np.random.normal(size=(len(X), 1))
        else:
            mu, sigma = self.gp.predict(X, return_std=True)
            lcb = mu - self.kappa * sigma
            return lcb.reshape(-1, 1)

    def _diversity_metric(self, X):
        if self.X is None:
            return np.ones((len(X), 1))
        else:
            distances = np.min(cdist(X, self.X), axis=1)
            return distances.reshape(-1, 1)

    def _select_next_points(self, batch_size):
        n_candidates = min(100 * self.dim, self.budget - self.n_evals)
        candidates = self._sample_points(n_candidates)

        ei = self._expected_improvement(candidates)
        lcb = self._lower_confidence_bound(candidates)
        diversity = self._diversity_metric(candidates)

        ei_normalized = (ei - np.min(ei)) / (np.max(ei) - np.min(ei)) if np.max(ei) != np.min(ei) else np.zeros_like(ei)
        lcb_normalized = (lcb - np.min(lcb)) / (np.max(lcb) - np.min(lcb)) if np.max(lcb) != np.min(lcb) else np.zeros_like(lcb)
        diversity_normalized = (diversity - np.min(diversity)) / (np.max(diversity) - np.min(diversity)) if np.max(diversity) != np.min(diversity) else np.zeros_like(diversity)

        # Weighted combination of EI and LCB
        acquisition = self.ei_weight * ei_normalized - (1 - self.ei_weight) * lcb_normalized

        F = np.hstack([acquisition, diversity_normalized])

        is_efficient = np.ones(F.shape[0], dtype=bool)
        for i, c in enumerate(F):
            if is_efficient[i]:
                is_efficient[is_efficient] = np.any(F[is_efficient] >= c, axis=1)
                is_efficient[i] = True

        pareto_front = candidates[is_efficient]

        if len(pareto_front) > 0:
            acquisition_pareto = self.ei_weight * self._expected_improvement(pareto_front) - (1-self.ei_weight) * self._lower_confidence_bound(pareto_front)
            next_point = pareto_front[np.argmax(acquisition_pareto)].reshape(1, -1)
        else:
            next_point = self._sample_points(1)

        return next_point

    def _evaluate_points(self, func, X):
        y = np.array([func(x) for x in X]).reshape(-1, 1)
        self.n_evals += len(X)
        return y

    def _update_eval_points(self, new_X, new_y):
        if self.X is None:
            self.X = new_X
            self.y = new_y
        else:
            self.X = np.vstack((self.X, new_X))
            self.y = np.vstack((self.y, new_y))

        best_index = np.argmin(self.y)
        if self.y[best_index][0] < self.best_y:
            self.best_y = self.y[best_index][0]
            self.best_x = self.X[best_index]
            self.success_ratio = 1.0
        else:
            self.success_ratio *= 0.75

    def _adjust_trust_region(self):
        if self.success_ratio > 0.5:
            self.trust_region_radius = min(self.trust_region_radius * self.radius_increase, 5.0)
        else:
            self.trust_region_radius = max(self.trust_region_radius * self.radius_decay, self.min_radius)

    def _adjust_kappa(self):
        self.kappa *= self.kappa_decay

    def _adjust_ei_weight(self):
        self.ei_weight = min(self.ei_weight * self.ei_weight_increase, 1.0)

    def __call__(self, func: Callable[[np.ndarray], np.float64]) -> tuple[np.float64, np.array]:
        initial_X = self._sample_points(self.n_init)
        initial_y = self._evaluate_points(func, initial_X)
        self._update_eval_points(initial_X, initial_y)

        batch_size = min(1, self.dim)
        while self.n_evals < self.budget:
            self._fit_model(self.X, self.y)
            next_X = self._select_next_points(batch_size)
            next_y = self._evaluate_points(func, next_X)
            self._update_eval_points(next_X, next_y)
            self._adjust_trust_region()
            self._adjust_kappa()
            self._adjust_ei_weight()

        return self.best_y, self.best_x
```
